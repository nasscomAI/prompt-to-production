skills:
  - name: retrieve_policy
    description: "loads .txt policy file, returns content as structured numbered sections"
    input: "file path (string) to the .txt policy document"
    output: "structured text or list of clauses representing the policy content"
    error_handling: >
      If the file cannot be read, return an error. Do not skip document sections.

  - name: summarize_policy
    description: "takes structured sections, produces compliant summary with clause references"
    input: "structured content generated by retrieve_policy"
    output: "a summarized text resolving the policy keeping all constraints intact"
    error_handling: >
      If a clause is identified where conditions would drop or meaning would bleed, quote it verbatim and flag it. Never drop an entire clause.
